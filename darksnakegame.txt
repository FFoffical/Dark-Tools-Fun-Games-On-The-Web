<script>
const board = document.getElementById('gameBoard');
const scoreEl = document.getElementById('score');
const startBtn = document.getElementById('startBtn');

const rows = 20;
const cols = 20;
const totalCells = rows * cols;
let cells = [];
let snake = [];
let food = {};
let dir = {x:0, y:0};
let nextDir = {x:0, y:0};
let score = 0;
let interval;
let speed = 150;
let gameRunning = false;

// Setup board grid
board.style.gridTemplateRows = `repeat(${rows}, 1fr)`;
board.style.gridTemplateColumns = `repeat(${cols}, 1fr)`;
board.style.width = `${cols*18}px`;
board.style.height = `${rows*18}px`;

// Create cells
for (let i = 0; i < totalCells; i++) {
  const cell = document.createElement('div');
  cell.classList.add('cell');
  board.appendChild(cell);
  cells.push(cell);
}

function index(x, y) {
  return y * cols + x;
}

function draw() {
  cells.forEach(c => c.className = 'cell');
  snake.forEach(s => {
    if (s.x >= 0 && s.x < cols && s.y >= 0 && s.y < rows)
      cells[index(s.x, s.y)].classList.add('snake');
  });
  cells[index(food.x, food.y)].classList.add('food');
}

function placeFood() {
  let newFood;
  do {
    newFood = {
      x: Math.floor(Math.random() * cols),
      y: Math.floor(Math.random() * rows)
    };
  } while (snake.some(s => s.x === newFood.x && s.y === newFood.y));
  food = newFood;
}

function move() {
  dir = nextDir; // apply queued direction
  if (dir.x === 0 && dir.y === 0) return; // not moving yet

  const head = {x: snake[0].x + dir.x, y: snake[0].y + dir.y};

  // Wrap-around
  if (head.x < 0) head.x = cols - 1;
  if (head.x >= cols) head.x = 0;
  if (head.y < 0) head.y = rows - 1;
  if (head.y >= rows) head.y = 0;

  // Collision with self
  if (snake.some(s => s.x === head.x && s.y === head.y)) {
    gameOver();
    return;
  }

  snake.unshift(head);

  // Eat food
  if (head.x === food.x && head.y === food.y) {
    score++;
    scoreEl.textContent = `Score: ${score}`;
    placeFood();
  } else {
    snake.pop();
  }

  draw();
}

function startGame() {
  clearInterval(interval);
  snake = [{x: 10, y: 10}];
  dir = {x: 1, y: 0}; // start moving right automatically
  nextDir = {...dir};
  score = 0;
  scoreEl.textContent = 'Score: 0';
  placeFood();
  draw();
  interval = setInterval(move, speed);
  gameRunning = true;
}

function gameOver() {
  clearInterval(interval);
  gameRunning = false;
  scoreEl.textContent = `üíÄ Game Over ‚Äî Score: ${score}`;
}

// Direction control
document.addEventListener('keydown', e => {
  if (!gameRunning) return;
  let newDir = {...dir};
  if (e.key === 'ArrowUp' && dir.y !== 1) newDir = {x: 0, y: -1};
  else if (e.key === 'ArrowDown' && dir.y !== -1) newDir = {x: 0, y: 1};
  else if (e.key === 'ArrowLeft' && dir.x !== 1) newDir = {x: -1, y: 0};
  else if (e.key === 'ArrowRight' && dir.x !== -1) newDir = {x: 1, y: 0};
  nextDir = newDir;
});

// Touch controls
let startX, startY;
board.addEventListener('touchstart', e => {
  const touch = e.touches[0];
  startX = touch.clientX;
  startY = touch.clientY;
});
board.addEventListener('touchend', e => {
  if (!gameRunning) return;
  const touch = e.changedTouches[0];
  const dx = touch.clientX - startX;
  const dy = touch.clientY - startY;
  if (Math.abs(dx) < 20 && Math.abs(dy) < 20) return; // ignore small swipes
  if (Math.abs(dx) > Math.abs(dy)) {
    if (dx > 0 && dir.x !== -1) nextDir = {x: 1, y: 0};
    else if (dx < 0 && dir.x !== 1) nextDir = {x: -1, y: 0};
  } else {
    if (dy > 0 && dir.y !== -1) nextDir = {x: 0, y: 1};
    else if (dy < 0 && dir.y !== 1) nextDir = {x: 0, y: -1};
  }
});

startBtn.addEventListener('click', startGame);
draw();
</script>
