<script>
const slider = document.getElementById('volume');
const muteBtn = document.getElementById('muteBtn');
const valueLabel = document.getElementById('volValue');
let lastVolume = 50;
let muted = false;

// update track fill color
function updateSlider(){
  const val = slider.value;
  slider.style.background = `linear-gradient(90deg, var(--accent) ${val}%, var(--track) ${val}%)`;
  valueLabel.textContent = `${val}%`;
  if(val == 0){
    muteBtn.textContent = 'üîá';
    muted = true;
  } else {
    muteBtn.textContent = 'üîä';
    muted = false;
  }
}

slider.addEventListener('input', updateSlider);

// mute/unmute toggle
muteBtn.addEventListener('click', ()=>{
  if(muted || slider.value == 0){
    slider.value = lastVolume || 50;
    muted = false;
  } else {
    lastVolume = slider.value;
    slider.value = 0;
    muted = true;
  }
  updateSlider();
});

updateSlider();
</script>
